\section{Background}

\subsection{GP setup}
Assume $(x_i,y_i)\in \R^d \times \{0,1\}$ observations come from:
$$
y_i \mid f, x \sim \mathrm{Bern}\bigl(\sigma(f(x_i))\bigr)
\quad\text{independently}$$
$$f \sim GP(0, K_\theta),$$
Our goal is to learn hyperparameters $\theta$, then compute the predictive mean and variance. \\
Throughout we use our equispaced fourier feature basis $K \approx \Phi \Phi^* = FD^2F^*$.

\subsection{Polyagamma Augmentation}
Introduce auxiliary 
$$\omega_i \sim PG(1,0)$$.
Use two identities: \begin{enumerate}
\item $\frac{(e^{u})^y}{(1+e^u)} = \frac{1}{2} \exp((y-\frac{1}{2})u) \int \exp(-\frac{1}{2} \omega u^2) PG(\omega| 1,0) d\omega $
    \item $\exp\bigl[-\tfrac{1}{2}\,\omega\,c^2\bigr]\;PG(\omega\mid1,0)
  \propto PG(\omega\mid 1,c)$%
\end{enumerate}

\begin{align}
    p(y,f)
    &= p(f)\prod_{i=1}^n \frac{1}{2}
       \exp\bigl[(y_i - 1/2)\,f(x_i)\bigr]
       \int \exp\Bigl[-\tfrac{1}{2}\,\omega_i\,f(x_i)^2\Bigr]\,
       PG(\omega_i\mid 1,0)\,d\omega_i 
    \\
    &\implies
    p(f,y,\omega)\;\propto\;
    p(f)\,\prod_{i=1}^n
      \exp\bigl[(y_i - 1/2)\,f(x_i)\bigr]
      \exp\Bigl[-\tfrac{1}{2}\,\omega_i\,f(x_i)^2\Bigr]
      PG(\omega_i\mid 1,0).\label{joint}
\end{align}
$$
\boxed{\omega_i \mid f,y \sim PG(1,f(x_i)) \text{ independently}},$$

$$
\boxed{f\mid \omega,y \sim \mathcal{N}(m, V)},$$
where 
$$
V = \bigl(K_{\theta}^{-1} + \Omega\bigr)^{-1},
\quad
m = V\,(\,y - \tfrac12\,).
$$

\section{E step}
Fix Kernel hyperparameters $\theta$; update variational distribution $q(\omega,f)$.
\subsection{Warmup; CAVI Updates}
Use mean-field variational family; posit:
$$
q(\omega, f) = q(f)\,\prod_{i=1}^n q(\omega_i).
$$

Optimize $q$ via CAVI:

\begin{enumerate} 
\item First: \textbf{Fix} $q(\omega)$.
\begin{align}
  \arg\max_{q(f)}
    \E_{q(f,\omega)}\log\frac{p(y,f,\omega)}{q(f,\omega)}
  &=
  \arg\max_{q(f)} \E_{q(f)}\,\E_{q(\omega)}
    \log\frac{p(f\mid y,\omega)}{q(f)}
  \\ 
  &=
  \arg\min_{q(f)} \mathrm{KL}\bigl(q(f)\,\|\,p(f\mid y,\,\E_{q(\omega)}[\omega])\bigr) \label{qf_exp}
  \\
  &=
  \mathcal{N}\bigl(\,\Sigma\,(y-\tfrac12),\,(K_\theta^{-1} + \Lambda)^{-1}\bigr),
\end{align}
where $\Sigma=(K_\theta^{-1} + \Lambda)^{-1}$.  Ie, the solution is the posterior
$p(f\mid y,\omega)$ evaluated at $\omega_i=\E_{q(\omega)}[\omega_i]$.  
\footnote{Can also show with $\beta\sim N(0,I)$ that we could get
  $q(\beta)=\mathcal N\bigl(m,\,(I+\Phi^*\Lambda\Phi)^{-1}\bigr)$.}

We reach $\ref{qf_exp}$ because $\log p(f|y,\omega)$ is linear in $\omega$:
\begin{align}
  \log p(f(x)\mid y,\omega)
  &=
  -\tfrac12\,(f(x)-\Sigma(y-\tfrac12))^T\,\Sigma^{-1}\,(f(x)-\Sigma(y-\tfrac12))
  + \mathrm{const}
  \\
  &=
  -\tfrac12\,f(x)^T\,(K_\theta^{-1}+\Omega)\,f(x) + \dots
\end{align}
So
$\E_{q(f)}\log p(f\mid y,\omega)=p\bigl(f\mid y,\E_q[\omega]\bigr)$.

\item \textbf{Fix} $q(f)$.  Recall that we can write $PG(\omega_i | 1,c_i) = \exp(-\frac{1}{2} \omega_i c^2) PG(\omega_i|1,0)$. This property of PG yields:
\begin{align}
  \arg\min_{q(\omega)}
    \E_{q(\omega)}\,\E_{q(f)}
    \log\frac{PG(\omega_i\mid 1,f(x_i))}{q(\omega_i)}
  &=
  \arg\min_{q(\omega)} \mathrm{KL}\bigl(q(\omega)\,\|\,PG(\omega_i\mid1,c_i)\bigr),
\end{align}
where $c_i = \sqrt{\E_{q(f)} f(x_i)^2} = \sqrt{\Var_{q(f)}[f(x_i)] + \bigl(\E_{q(f)}[f(x_i)]\bigr)^2}$
So we have
$$
q(\omega_i)=PG(\omega_i\mid 1,\,c_i),
$$
\end{enumerate}
\subsection{CAVI Summary}
$$
\boxed{q(f)=\mathcal{N}(\mu,\Sigma)},\qquad
\boxed{q(\omega_i)\sim PG(1,c_i)}
$$
where we’ve defined:
\[
c_i=\sqrt{\Var_{q(f)}[f(x_i)] + (\E_{q(f)}[f(x_i)])^2},\quad
\Lambda=\diag\!\bigl(\E_q[\omega_i]\bigr)
       =\diag\!\bigl(\tfrac{\tanh(c_i/2)}{2c_i}\bigr),
\quad
\Sigma=(K_\theta^{-1}+\Lambda)^{-1},

\]

\eta_2^{(t)}

Now consider the $\eta_2$ update. Wenzel introduces $m$ inducing points, so that $\eta_2\in\mathbb R^{m\times m}$. However, for non-sparse GP ($m=n$), then storing and updating an $n\times n$ matrix is impractical. But for the $M$-step, all we have to store is how to apply $\Sigma$ The proposed solution is to first rewrite:


\[
-2\eta_2^{(t)} = \Sigma_{(t)}^{-1}
  =K_{\theta^{(t)}}^{-1} + \Delta^{(t)}
  \leftarrow
  (1-\rho)\bigl(K_{\theta^{(t-1)}}^{-1}+\Delta^{(t-1)}\bigr)
  +\rho\bigl(K_{\theta^{(t)}}^{-1}+\tfrac{n}{s}I_S^T\Lambda_S I_S\bigr),
\]
Instead of updating $\Sigma$, update:
\[
\Delta^{(t)}
  \leftarrow (1-\rho)\,\Delta^{(t-1)}
  +\rho\,\tfrac{n}{s}\,I_S^T\Lambda_S I_S.
\]

then either 

\begin{enumerate}
    \item Store the function:

$$v\mapsto \Sigma^{-1}_{(t)} v = (1-\rho)K_{\theta^{(t-1)}}^{-1}v + \rho(K_{\theta^{(t)}}^{-1} v) + \Delta^{(t)}v,$$

for use in the $M$-step. Or: \item Run enough updates of natural parameters before moving to the M-step that $(1-\rho)^k$ is negligible.\\
Or: \item Simply "ignore" the old kernel; after M-step, reset $\Sigma^{(t-1)} = K^{-1}_{\theta^{(t)}} + \Delta^{(t-1)}$. 
\end{enumerate}

I think the last option is most practical, if it works. We avoid applying this convex mix of $K^{-1}$ operators in the M-step, which would be costly and confusing. Then since $\theta$ will be constant within the $E$-step, we can just update $\Delta$ and then store the apply $v\mapsto \Sigma_{(t)}^{-1}v = (K_\theta^{-1} + \Delta^{(t)})v$.

\subsection{Solving $z\mapsto \Sigma z$}
Solve for $\alpha$:
\textbf{Version 1: $n\times n$ system}
\[
\Sigma z=(K^{-1}+\Delta)^{-1}z=\alpha
\;\iff\;
(I + K\Delta)\,\alpha = K z
\;\iff\;
(I + FD^2 F^* \Delta)\,\alpha = FD^2 F^* z.
\]
\textbf{Version 2: $m\times m$ system}:

\begin{align}
(K^{-1} + \Delta) v = z &\iff (I + K\Delta)v = Kz\\
&\iff (I + \Phi \Phi^* \Delta) v =  \Phi \Phi^*z \\
&\iff (I+ \Phi \Phi^* \Delta) \Phi u = \Phi \Phi^* z\label{image}\\
&\iff \Phi \left[ u + \Phi^* \Delta \Phi u\right] = \Phi\Phi^* z\\
&\iff (I + \Phi^* \Delta \Phi) u =\Phi^* z \text{ solve for u} \label{cancel}\\
&\text{ solve the above for $u$ and substitute } v = \Phi u.
\end{align}
Note that in \ref{image}, we can rearrange and see that $v \in \operatorname{Range}(\Phi)$. Also in \ref{cancel}, we can "cancel" the $\Phi$ on both sides since $\Phi$ is injective, since $\{\exp(i\omega_jx)\}_{j\in [M]}$ are linearly independent. 
\paragraph{Stable feature-space path actually used in the E-step.}
The derivation above gives the direct identity
\[
\boxed{
\Sigma z
=
\Phi (I + \Phi^* \Delta \Phi)^{-1}\Phi^* z.
}
\]
This is the numerically stable version of the E-step linear solve. In code, the feature-space system is
\[
A_E u = r_E,
\qquad
A_E := I + \Phi^* \Delta \Phi = I + D F^* \Delta F D,
\qquad
r_E := \Phi^* z = D F^* z,
\]
and once $u$ is solved for, we return
\[
\Sigma z = \Phi u = F D u.
\]

Previously we used the Woodbury-style reconstruction
\[
\Sigma
=
K - K(I+\Delta K)^{-1}\Delta K,
\]
which gives
\[
\Sigma z = Kz - K(I+\Delta K)^{-1}\Delta K z.
\]
If we write the correction term in feature space, this becomes
\[
q_E := \Phi^* \Delta K z,
\qquad
(I+\Phi^* \Delta \Phi)u_{\rm corr}=q_E,
\qquad
\Sigma z = Kz - \Phi u_{\rm corr}.
\]
So the \emph{operator} being inverted is the same in both formulations, but the right-hand side and reconstruction differ:
\begin{align*}
\text{old path:}\quad
&(I+\Phi^* \Delta \Phi)u_{\rm corr}=\Phi^* \Delta K z,
\qquad
\Sigma z = Kz - \Phi u_{\rm corr},\\
\text{new path:}\quad
&(I+\Phi^* \Delta \Phi)u=\Phi^* z,
\qquad
\Sigma z = \Phi u.
\end{align*}
At large $n$, the old path is unstable because $Kz$ and $\Phi u_{\rm corr}$ can each be very large while their difference $\Sigma z$ is much smaller, so small solve error is amplified by catastrophic cancellation. The new path avoids this subtraction entirely.

\begin{enumerate}
  \item In the direct feature-space formulation, each CG iteration applies $I + DF^*\Delta FD$ to the current feature vector.
  \item Plus two NUFFT setups.
  \item A warm start can be obtained by replacing $\Delta$ by $\bar\delta I$, solving the Toeplitz surrogate once, and using that approximate solution as the initial guess for the exact CG solve.
\end{enumerate}

\paragraph{Toeplitz preconditioner for the E-step solve.}
Write the feature-space linear system in the form
\[
A_E u = r_E,
\qquad
A_E := I + \Phi^* \Delta \Phi = I + D F^* \Delta F D,
\qquad
r_E := \Phi^* z = D F^* z.
\]
Let
\[
\bar\delta := \frac{1}{n}\sum_{i=1}^n \Delta_{ii}.
\]
If $\Delta$ is close to a scalar multiple of the identity, then a natural approximation is
\[
M_E := I + \bar\delta \Phi^* \Phi.
\]
Since $\Phi = F D$, this becomes
\[
M_E = I + \bar\delta D(F^*F)D.
\]
On the tensor grid used by the spectral approximation, the operator $F^*F$ is applied through the Toeplitz convolution operator $T$, so in code we use
\[
M_E \approx I + \bar\delta D T D.
\]
Instead of running plain CG on $A_E u = r_E$, we solve the left-preconditioned system
\[
M_E^{-1} A_E u = M_E^{-1} r_E.
\]
Each outer iteration still applies the exact operator $A_E$, so it still costs two NUFFT applications. The point is that the preconditioner application
\[
v \mapsto M_E^{-1} v
\]
is much cheaper, because it is computed by solving
\[
(I + \bar\delta D T D)x = v
\]
using only Toeplitz matvecs. In practice we do not form $M_E^{-1}$ explicitly; rather, each call to $M_E^{-1}v$ is approximated by an inner CG solve on the Toeplitz system above. Thus the implementation is an inexact left-preconditioned CG method whose expensive work remains in the outer NUFFT-backed iterations.

The older warm-start idea corresponds to solving
\[
M_E x_0 = r_E
\]
once and then using $x_0$ as the initial iterate for the exact solve. Preconditioning is stronger: it uses $M_E^{-1}$ at every outer iteration rather than only once at the beginning.

\paragraph{Toeplitz preconditioner for the M-step solve.}
The same idea applies to the whitened Hermitian feature-space system used in the revised $M$-step. There the linear operator is
\[
A_M := I + S,
\qquad
S = D^{1/2} F^* \Omega F D^{1/2},
\]
where $\Omega = \Delta$ is diagonal and $D^{1/2}$ denotes the diagonal matrix of square-root spectral weights. If we define
\[
\bar\omega := \frac{1}{n}\sum_{i=1}^n \Omega_{ii},
\]
then the Toeplitz surrogate is
\[
M_M := I + \bar\omega D^{1/2} T D^{1/2}.
\]
The preconditioned system is
\[
M_M^{-1} A_M y = M_M^{-1} r_M.
\]
Again, each outer iteration applies the exact NUFFT-backed operator $A_M$, while each preconditioner application is carried out by an inner Toeplitz-only solve
\[
(I + \bar\omega D^{1/2} T D^{1/2})x = v.
\]
This is the same operator that appears in the predictive-variance computation, so the same Toeplitz preconditioner can be reused there as well.

\paragraph{When is this useful?}
The approximation is best when $\Delta$ is reasonably close to $\bar\delta I$ (or, in the $M$-step, when $\Omega$ is close to $\bar\omega I$). In that regime, the preconditioner can cut the number of expensive outer NUFFT iterations dramatically. When the spectral system dimension $M$ is itself very large, the extra inner Toeplitz solves may not always be favorable, so this is best exposed as an optional solver mode rather than hard-coded as the only strategy.

\section{Revised M Step}
Use Woodbury inversion Lemma to get:

$$\Sigma = (K^{-1} + \Delta)^{-1} = K - K(I + \Delta K)^{-1} \Delta K.$$

Hence we have the identity:

$$K^{-1} \Sigma K^{-1} = K^{-1} - (I+\Delta K)^{-1} \Delta.$$
\begin{align}
\frac{\partial}{\partial\theta}\mathcal{L}
  &=\frac{\partial}{\partial\theta}
    \E_q\log p(f)
  \\
  &=\frac{\partial}{\partial\theta}
    \Bigl(
      -\tfrac12\,\tr\bigl((\Sigma + \mu\mu^T)K^{-1}\bigr)
      -\tfrac12\log|K|
    \Bigr)
  \\
  &=\tfrac12\,\mu^T K^{-1}(\partial_\theta K)\,K^{-1}\mu
    +\tfrac12\,\tr\bigl(\Sigma\,K^{-1}(\partial_\theta K)\,K^{-1}\bigr)
    -\tfrac12\,\tr\bigl(K^{-1}\partial_\theta K\bigr)\\
    & = I + II + III.
\end{align}

Cycling the traces:
\begin{align}\tfrac12\,\tr\bigl(\Sigma\,K^{-1}(\partial_\theta K)\,K^{-1}\bigr)&= \tfrac12\,\tr\bigl(K^{-1}\Sigma\,K^{-1}(\partial_\theta K)\bigr)\\
& = \frac{1}{2}\tr (K^{-1} \partial_\theta K) - \frac{1}{2} \tr ((I+\Delta K)^{-1} \Delta \partial_\theta K)
\end{align}

Hence we can cancel the third term and write:

\begin{align}\frac{\partial}{\partial \theta} \mathcal L &= I -\frac{1}{2} \tr((I+ \Delta K)^{-1} \Delta \partial_\theta K).\\
& = I - \frac{1}{2}\tr(\Delta FD'F^* (I+\Delta K)^{-1})
\end{align}

This is an improvement over before since that inverse is well-conditioned. The idea to solve it efficiently is to approximate $\Delta$ as constant times identity. Let $\alpha := (I+\Delta K)^{-1} z.$ Then     $z = (I + \Delta FD^2)F^* \alpha$ and:

    &\approx (I + \bar \delta TD^2) F^* \alpha.
\end{align}

Solve the second system using Toeplitz machinery, then warm start the above system--should only take a  few iterations to converge with this warm start.

\section{Predictive mean}
As in Wenzel:

\begin{align}
    p(f_*|y) &= \int p(f_*|f) p(f|y) df\\
    & \approx \int p(f_*|f) q(f) df\\
    & = N(f_*|\mu_*, \sigma_*^2),
\end{align}
where it is clearer to work in feature space. Let
\[
\Psi := FD,
\qquad
K=\Psi\Psi^*=FD^2F^*,
\qquad
\kappa := y-\tfrac12,
\qquad
B := \Psi^*\Delta \Psi = D F^* \Delta F D.
\]
Here:
\[
F\in \mathbb C^{n\times M}
\]
is the Fourier / NUFFT feature matrix at the training inputs,
\[
D=\diag(ws)\in\mathbb R^{M\times M}
\]
is the diagonal matrix of spectral weights, and
\[
\Delta=\diag(\delta)\in\mathbb R^{n\times n}
\]
is the diagonal PolyGamma variational parameter matrix.
The Gaussian variational factor is
\[
q(f)=N(\mu,\Sigma),
\qquad
\Sigma=(K^{-1}+\Delta)^{-1},
\qquad
\mu = \Sigma \kappa.
\]
Now define the feature-space mean coefficients by
\[
m_u := (I+B)^{-1}\Psi^* \kappa \in \mathbb C^M,
\]
and the candidate training-location mean
\[
\tilde{\mu} := \Psi m_u.
\]
We now show that $\tilde{\mu}=\mu$.
First,
\[
\mu
  = (K^{-1}+\Delta)^{-1}\kappa.
\]
Using
\[
(K^{-1}+\Delta)^{-1}=(I+K\Delta)^{-1}K,
\]
we get
\[
\mu = (I+K\Delta)^{-1}K\kappa.
\]
Next, since $K=\Psi\Psi^*$ and
\[
(I+AB)^{-1}A = A(I+BA)^{-1},
\]
with
\[
A=\Psi,
\qquad
B=\Psi^*\Delta,
\]
we obtain
\[
(I+K\Delta)^{-1}\Psi
  =
  (I+\Psi\Psi^*\Delta)^{-1}\Psi
  =
  \Psi(I+\Psi^*\Delta\Psi)^{-1}.
\]
Therefore
\[
(I+K\Delta)^{-1}K
  =
  (I+K\Delta)^{-1}\Psi\Psi^*
  =
  \Psi(I+\Psi^*\Delta\Psi)^{-1}\Psi^*
  =
  \Psi(I+B)^{-1}\Psi^*.
\]
Substituting this into the expression for $\mu$ gives
\[
\mu
  =
  \Psi(I+B)^{-1}\Psi^*\kappa
  =
  \Psi m_u
  =
  \tilde{\mu}.
\]
Hence the posterior mean at the training locations is indeed
\[
\mu = \Psi m_u.
\]

Equivalently, the posterior mean satisfies
\[
\left(K^{-1}+\Delta\right)\mu=\kappa,
\qquad
K^{-1}\mu = \kappa-\Delta\mu.
\]
Applying $\Psi^*$ gives
\[
\Psi^*K^{-1}\mu = \Psi^*(\kappa-\Delta\mu).
\]
Since $\mu=\Psi m_u$, this becomes
\[
\Psi^*K^{-1}\mu = \Psi^*\kappa-\Psi^*\Delta\Psi m_u.
\]
But from
\[
(I+B)m_u=\Psi^*\kappa,
\]
we also have
\[
m_u = \Psi^*\kappa-\Psi^*\Delta\Psi m_u.
\]
Comparing the last two displays shows that
\[
m_u = \Psi^*K^{-1}\mu.
\]
Multiplying by $D^{-1}$ and using $\Psi=FD$ yields the equivalent relation
\[
\beta^{(1)} := F^*K^{-1}\mu,
\qquad
m_u = D\beta^{(1)}.
\]

For a new point $x_*$, define the unweighted Fourier row
\[
(\varphi_*)_j
  :=
  \exp(2\pi i \langle \xi_j, x_*\rangle),
  \qquad j\in[M],
\]
and the weighted feature vector
\[
(\psi_*)_j
  :=
  \sqrt{h^d \hat{k}(\xi_j)}\,
  \exp(2\pi i \langle \xi_j, x_*\rangle),
  \qquad j\in[M].
\]
Equivalently,
\[
\psi_* = D\varphi_*.
\]
The cross-covariance vector between the training inputs and $x_*$ is
\[
k_* := K(X,x_*) = \Psi\psi_* = FD^2\varphi_*.
\]
Then the predictive latent mean is
\[
\mu_*
  = \E[f_*|y]
  \approx k_*^T K^{-1}\mu.
\]
Using the identities above,
\begin{align}
\mu_*
  &= k_*^T K^{-1}\mu\\
  &= \psi_*^* \Psi^* K^{-1}\mu\\
  &= \psi_*^* m_u.
\end{align}
Hence
\[
\boxed{
\mu_* = \psi_*^* m_u.
}
\]


To connect back to Term 1 in the M-step, define
\[
\beta^{(1)} := F^*K^{-1}\mu,
\]
Therefore
\[
\mu_* = \varphi_*^* D m_u = \varphi_*^* D^2 \beta^{(1)}.
\]
In practice, for a batch of test locations this is a single Type-2 NUFFT.

\section{Predictive variance}
Starting from the conditional GP variance formula and integrating against
$q(f)=N(\mu,\Sigma)$ gives
\[
\sigma_*^2
  =
  k_{**}
  - k_*^T K^{-1}k_*
  + k_*^T K^{-1}\Sigma K^{-1}k_*.
\]
Using the revised M-step identity
\[
K^{-1}\Sigma K^{-1}
  =
  K^{-1} - (I+\Delta K)^{-1}\Delta,
\]
the $K^{-1}$ terms cancel, so
\[
\boxed{
\sigma_*^2
  =
  k_{**}
  - k_*^T (I+\Delta K)^{-1}\Delta k_*.
}
\]
Thus we never need to store or apply $K^{-1}$ explicitly.

In the feature notation above, let
\[
S_u := (I+B)^{-1}.
\]
Since $\Sigma = \Psi S_u \Psi^*$, the predictive variance can be written as
\[
\boxed{
\sigma_*^2
  =
  \psi_*^* S_u \psi_*
  =
  \psi_*^*(I+B)^{-1}\psi_*.
}
\]
So predictive variances reuse the same Hermitian operator
$I + D F^* \Delta F D$ as the revised M-step: for each test point, solve
\[
(I+B)u_* = \psi_*,
\]
then compute
\[
\sigma_*^2 = \psi_*^* u_*.
\]
For a batch of test points, solve these systems simultaneously with batched CG.
\subsection{Stochastic predictive variance via diagonal sums}
The exact formula above is still expensive for large numbers of test points,
because it requires solving the same linear system against many different
right-hand sides. In the PolyGamma case we can reuse the same diagonal-sum idea
from the conjugate setting, because the predictive variance is still a
quadratic form in the Fourier features.

Write
\[
A := I + B = I + D F^* \Delta F D,
\qquad
G := D A^{-1} D.
\]
If $\varphi_*$ denotes the unweighted Fourier row at $x_*$ and
$\psi_* = D\varphi_*$, then
\[
\sigma_*^2
  =
  \psi_*^* A^{-1}\psi_*
  =
  \varphi_*^* G \varphi_*.
\]
Expanding this entrywise gives
\[
\sigma_*^2
  =
  \sum_{j=1}^{M^d}\sum_{\ell=1}^{M^d}
  e^{2\pi i \langle \xi_\ell - \xi_j, x_* \rangle}\,G_{j\ell}.
\]
Because the frequencies lie on a tensor grid, $\xi_j = h m_j$ with
$m_j \in \{-(M-1)/2,\ldots,(M-1)/2\}^d$, the differences
$\xi_\ell - \xi_j$ also lie on a grid. Therefore
\[
\sigma_*^2
  =
  \sum_{r \in \{-(M-1),\ldots,M-1\}^d}
  e^{2\pi i h \langle x_*, r \rangle}\, c[r],
\]
where the offset-diagonal sums are
\[
\boxed{
c[r]
  :=
  \sum_{j=1}^{M^d}\sum_{\ell=1}^{M^d}
  G_{j\ell}\,1[m_\ell - m_j = r].
}
\]
Once the coefficients $c[r]$ are known, all predictive variances on a test set
are obtained by a single Type-2 NUFFT on the $(2M-1)^d$ grid of offsets.

We estimate $c[r]$ stochastically. Let
$\eta^{(s)} \in \{-1,1\}^{M^d}$ be i.i.d.\ Rademacher probes. For each probe,
solve
\[
A \beta^{(s)} = D \eta^{(s)},
\]
using exactly the same feature-space CG solver as in the $E$-step and revised
$M$-step, with the same Toeplitz warm start or Toeplitz preconditioner if
desired. Define
\[
\gamma^{(s)} := D \beta^{(s)}.
\]
Since $\beta^{(s)} = A^{-1} D \eta^{(s)}$, we have
\[
\gamma^{(s)} = D A^{-1} D \eta^{(s)} = G \eta^{(s)}.
\]
For any fixed offset $r$,
\[
\widehat c^{(s)}[r]
  :=
  \sum_j \gamma^{(s)}_j \eta^{(s)}_{j+r}
\]
is an unbiased estimator of $c[r]$, because
\[
\E[\gamma^{(s)}_j \eta^{(s)}_\ell]
  =
  \E[(G\eta^{(s)})_j \eta^{(s)}_\ell]
  =
  \sum_k G_{jk}\E[\eta^{(s)}_k \eta^{(s)}_\ell]
  =
  G_{j\ell}.
\]
Hence
\[
\E[\widehat c^{(s)}[r]]
  =
  \sum_j G_{j,j+r}
  =
  c[r].
\]
Averaging over $J$ probes gives
\[
\widehat c[r]
  =
  \frac{1}{J}\sum_{s=1}^J \widehat c^{(s)}[r].
\]

The computation of $\widehat c[r]$ is itself efficient. Reshape
$\gamma^{(s)}$ and $\eta^{(s)}$ to $M \times \cdots \times M$ tensors. Then
$\widehat c^{(s)}[r]$ is a cross-correlation, so all offsets are obtained by
FFT:
\[
\widehat c^{(s)}
  =
  \mathcal F^{-1}\!\left(
    \mathcal F(\gamma^{(s)}) \odot \overline{\mathcal F(\eta^{(s)})}
  \right),
\]
zero-padded to shape $(2M-1,\ldots,2M-1)$.

This gives the following approximation scheme for predictive variances:
\begin{enumerate}
    \item Draw $J$ probe vectors $\eta^{(s)}$.
    \item Solve $A\beta^{(s)} = D\eta^{(s)}$ for each probe.
    \item Form $\gamma^{(s)} = D\beta^{(s)}$ and estimate all $c[r]$ by FFT
    cross-correlations.
    \item Evaluate $\sigma_*^2 \approx \sum_r e^{2\pi i h \langle x_*, r \rangle}\widehat c[r]$
    at all test points by one Type-2 NUFFT.
\end{enumerate}
The dominant cost is now the $J$ probe solves, rather than one solve per test
block. For large test sets this is much cheaper, since $J$ is chosen
independently of the number of test points.
\section{Predictive Class Probability }
Finally, the predictive class probability is
\[
p(y_* = 1|y)
  =
  \int \sigma(f_*)\,p(f_*|y)\,df_*
  \approx
  \int \sigma(f_*)\,N(f_*|\mu_*,\sigma_*^2)\,df_*,
\]
which is a one-dimensional integral. One may evaluate it by quadrature, or use the same approximation as at the training points:
\[
p(y_* = 1|y)
  \approx
  \sigma\!\left(
    \frac{\mu_*}{\sqrt{1+\pi\sigma_*^2/8}}
  \right).
\]

\section{Summary}
We can learn hyperparameters, compute predictive means in $\calO(n + M\log M)$, but the cost of applying $K^{-1} + \Delta$ will give a larger constant than in the conjugate case, since we must rely on NUFFT in conjugate gradient solves, rather than the Toeplitz operation. \\
\section{Negative Binomial}
We can use the same PolyGamma augmentation for negative binomial observations. The key point is that the Fourier / NUFFT linear algebra is unchanged; only the likelihood-dependent scalars in the $E$-step change, and if we choose to learn the shape parameter $r$, that is an additional scalar update in the $M$-step.

\subsection{Likelihood and augmentation}
Let
\[
y_i \mid f_i, r
  \sim \mathrm{NB}(r,\sigma(f_i)),
\]
with pmf
\[
p(y_i\mid f_i,r)
  =
  \binom{y_i+r-1}{y_i}\,
  \sigma(f_i)^{y_i}\,
  \bigl(1-\sigma(f_i)\bigr)^r.
\]
Using
\[
\sigma(f)=\frac{e^f}{1+e^f},
\qquad
1-\sigma(f)=\frac{1}{1+e^f},
\]
the $f_i$-dependent part is
\[
p(y_i\mid f_i,r)
  \propto
  \frac{e^{y_i f_i}}{(1+e^{f_i})^{y_i+r}}.
\]
Apply the PolyGamma identity with
\[
a=y_i,
\qquad
b=y_i+r,
\qquad
\kappa_i := a-\frac{b}{2} = \frac{y_i-r}{2}.
\]
Then
\[
\frac{e^{y_i f_i}}{(1+e^{f_i})^{y_i+r}}
  =
  2^{-(y_i+r)}
  e^{\kappa_i f_i}
  \int_0^\infty
    \exp\!\Bigl(-\frac{1}{2}\omega_i f_i^2\Bigr)\,
    PG(\omega_i\mid y_i+r,0)\,d\omega_i.
\]
Hence the augmented joint is
\[
p(f,y,\omega\mid r,\theta)
  =
  p(f\mid\theta)
  \prod_{i=1}^n
  \binom{y_i+r-1}{y_i}\,
  2^{-(y_i+r)}\,
  e^{\kappa_i f_i}\,
  e^{-\frac{1}{2}\omega_i f_i^2}\,
  PG(\omega_i\mid y_i+r,0).
\]

\subsection{Mean-field updates}
Use the same variational family
\[
q(f,\omega)=q(f)\prod_{i=1}^n q(\omega_i).
\]
Conditioning on $q(\omega)$ gives
\[
\log q^*(f)
  =
  -\frac{1}{2}f^T K_\theta^{-1} f
  + \kappa^T f
  - \frac{1}{2}f^T \Lambda f
  + \mathrm{const},
\]
where
\[
\kappa
  :=
  \left(\frac{y_1-r}{2},\ldots,\frac{y_n-r}{2}\right)^T,
\qquad
\Lambda
  :=
  \diag\!\bigl(\E_q[\omega_i]\bigr).
\]
Therefore
\[
q(f)=N(\mu,\Sigma),
\qquad
\Sigma=(K_\theta^{-1}+\Lambda)^{-1},
\qquad
\mu=\Sigma\kappa.
\]
Fixing $q(f)$ and using the same tilting identity as in the Bernoulli case,
\[
q^*(\omega_i)=PG(y_i+r,c_i),
\]
where
\[
c_i
  :=
  \sqrt{\E_q[f_i^2]}
  =
  \sqrt{\mu_i^2 + \Sigma_{ii}}.
\]
Its mean is
\[
\E_q[\omega_i]
  =
  \frac{y_i+r}{2c_i}\tanh\!\Bigl(\frac{c_i}{2}\Bigr).
\]
Thus the only changes relative to Bernoulli are
\[
\kappa_i = \frac{y_i-r}{2},
\qquad
b_i = y_i+r,
\qquad
\Lambda_{ii}
  =
  \frac{y_i+r}{2c_i}\tanh\!\Bigl(\frac{c_i}{2}\Bigr).
\]
All of the Fourier-feature machinery from the Bernoulli model is unchanged:
\begin{enumerate}
  \item $\Sigma z = (K^{-1}+\Lambda)^{-1} z$ is still applied with the same $M\times M$ Hermitian solve,
  \item $\diag(\Sigma)$ is still estimated by Hutchinson probes,
  \item the Toeplitz warm start still uses the approximation $\Lambda \approx \bar{\lambda} I$.
\end{enumerate}

\subsection{Kernel hyperparameter update}
For fixed $q(f)$, the kernel hyperparameter gradient is exactly the same as in the Bernoulli case, because only the prior term depends on $\theta$:
\[
\frac{\partial}{\partial \theta}\mathcal L
  =
  \frac{\partial}{\partial \theta}
  \E_q[\log p(f\mid \theta)].
\]
Therefore the revised $M$-step formula from the previous section applies without modification:
\[
\frac{\partial}{\partial \theta}\mathcal L
  =
  \frac{1}{2}\mu^T K^{-1}(\partial_\theta K)K^{-1}\mu
  -
  \frac{1}{2}
  \Tr\!\bigl((I+\Lambda K)^{-1}\Lambda\,\partial_\theta K\bigr).
\]
So the same Term 1 / trace-estimation machinery, NUFFTs, and warm-started CG are reused.

\subsection{Updating the shape parameter $r$}
Now suppose $r$ is also learned. For fixed $q(f)$, the only part of the ELBO that depends on $r$ is the expected log-likelihood:
\[
\mathcal L(r)
  =
  \sum_{i=1}^n
  \E_{q(f_i)}\bigl[\log p(y_i\mid f_i,r)\bigr]
  + \mathrm{const}.
\]
Using
\[
\log p(y_i\mid f_i,r)
  =
  \log\Gamma(y_i+r)
  - \log\Gamma(r)
  - \log(y_i!)
  + y_i f_i
  - (y_i+r)\log(1+e^{f_i}),
\]
we differentiate with respect to $r$ while holding $q(f)$ fixed:
\[
\frac{\partial \mathcal L}{\partial r}
  =
  \sum_{i=1}^n
  \left[
    \psi(y_i+r)-\psi(r)
    -
    \E_{q(f_i)}\log(1+e^{f_i})
  \right].
\]
Since
\[
\log \sigma(-f) = -\log(1+e^f),
\]
this is equivalently
\[
\boxed{
\frac{\partial \mathcal L}{\partial r}
  =
  \sum_{i=1}^n
  \left[
    \psi(y_i+r)-\psi(r)
    +
    \E_{q(f_i)}\log \sigma(-f_i)
  \right].
}
\]
Under the Gaussian variational approximation
\[
q(f_i)=N(\mu_i,\Sigma_{ii}),
\]
the remaining expectation is a one-dimensional Gaussian integral:
\[
\E_{q(f_i)}\log \sigma(-f_i)
  =
  \int
    \log \sigma(-f)\,
    N(f\mid \mu_i,\Sigma_{ii})\,df.
\]
Writing $f=\mu_i+\sqrt{\Sigma_{ii}}\,t$ with $t\sim N(0,1)$ gives
\[
\E_{q(f_i)}\log \sigma(-f_i)
  =
  \int
    \log \sigma\!\bigl(-(\mu_i+\sqrt{\Sigma_{ii}}\,t)\bigr)\,
    N(t\mid 0,1)\,dt,
\]
which is evaluated numerically by Gauss--Hermite quadrature. Let $\{x_k,w_k\}_{k=1}^K$ be the standard Gauss--Hermite nodes and weights for the weight $e^{-x^2}$, so that
\[
\int_{-\infty}^{\infty} h(x)e^{-x^2}\,dx
  \approx
  \sum_{k=1}^K w_k h(x_k).
\]
Since
\[
\int g(t)\,N(t\mid 0,1)\,dt
  =
  \frac{1}{\sqrt{\pi}}
  \int
    g(\sqrt{2}\,x)e^{-x^2}\,dx,
\]
we obtain
\[
\E_{q(f_i)}\log \sigma(-f_i)
  \approx
  \frac{1}{\sqrt{\pi}}
  \sum_{k=1}^K
    w_k\,
    \log \sigma\!\Bigl(-\bigl(\mu_i+\sqrt{2\Sigma_{ii}}\,x_k\bigr)\Bigr).
\]
Equivalently, define normalized nodes and weights
\[
\tilde t_k := \sqrt{2}\,x_k,
\qquad
\tilde w_k := \frac{w_k}{\sqrt{\pi}},
\]
so that
\[
\boxed{
\E_{q(f_i)}\log \sigma(-f_i)
  \approx
  \sum_{k=1}^K
    \tilde w_k\,
    \log \sigma\!\Bigl(-\bigl(\mu_i+\sqrt{\Sigma_{ii}}\,\tilde t_k\bigr)\Bigr).
}
\]
Thus for each $i$ the actual computation is:
\begin{enumerate}
  \item form $\sigma_i := \sqrt{\Sigma_{ii}}$,
  \item evaluate the $K$ shifted points
  \[
  f_{ik} := \mu_i + \sigma_i \tilde t_k,
  \]
  \item compute
  \[
  \ell_{ik} := \log \sigma(-f_{ik}),
  \]
  \item take the weighted sum
  \[
  \sum_{k=1}^K \tilde w_k \ell_{ik}.
  \]
\end{enumerate}
Vectorizing over all $i\in[n]$, this is just one broadcasted evaluation of an $n\times K$ array
\[
f_{ik} = \mu_i + \sqrt{\Sigma_{ii}}\,\tilde t_k,
\]
followed by an $n\times K$ application of $\log \sigma(-\cdot)$ and a weighted reduction over $k$. The total cost is $\calO(nK)$ for fixed $K$, with no NUFFT, Toeplitz, or CG solve once $\mu_i$ and $\Sigma_{ii}$ are already available from the $E$-step.
Equivalently, if we stack these terms into a matrix
\[
F
  :=
  \begin{bmatrix}
    f_{11} & \cdots & f_{1K}\\
    \vdots & \ddots & \vdots\\
    f_{n1} & \cdots & f_{nK}
  \end{bmatrix}
  \in \R^{n\times K},
\]
then the quadrature contribution to the $r$-gradient is
\[
g_i
  :=
  \sum_{k=1}^K \tilde w_k \log \sigma(-f_{ik}),
\qquad
g=(g_1,\ldots,g_n)^T\in\R^n.
\]
So the work is:
\begin{enumerate}
  \item form the $n\times K$ matrix $F$ : $\calO(nK)$,
  \item apply $\log \sigma(-\cdot)$ entrywise to $F$ : $\calO(nK)$,
  \item multiply by the $K$ weights and sum across columns : $\calO(nK)$.
\end{enumerate}
Hence the whole quadrature step is linear in the number of observations $n$, up to the fixed quadrature size $K$.

To enforce positivity, we optimize
\[
\eta := \log r.
\]
By the chain rule,
\[
\frac{\partial \mathcal L}{\partial \eta}
  =
  \frac{\partial \mathcal L}{\partial r}
  \frac{\partial r}{\partial \eta}
  =
  r\,
  \frac{\partial \mathcal L}{\partial r}.
\]
Hence a gradient-ascent update takes the form
\[
\eta^{(t+1)}
  =
  \eta^{(t)}
  +
  \rho_r\,
  r^{(t)}
  \sum_{i=1}^n
  \left[
    \psi(y_i+r^{(t)})-\psi(r^{(t)})
    +
    \E_{q(f_i)}\log \sigma(-f_i)
  \right].
\]
After updating $r$, the next $E$-step uses the refreshed quantities
\[
\kappa_i = \frac{y_i-r}{2},
\qquad
b_i = y_i+r,
\qquad
\Lambda_{ii}
  =
  \frac{y_i+r}{2c_i}\tanh\!\Bigl(\frac{c_i}{2}\Bigr).
\]
So learning $r$ changes the next diagonal system through $\kappa$ and $\Lambda$, but the expensive linear algebra is exactly the same as before.

\subsection{Predictive mean count}
In this parameterization,
\[
\E[y_i\mid f_i,r] = r e^{f_i}.
\]
Therefore, under the Gaussian approximation
\[
f_* \mid y \approx N(\mu_*,\sigma_*^2),
\]
the predictive mean count is
\[
\E[y_*\mid y]
  \approx
  r\,\E[e^{f_*}]
  =
  \boxed{
  r\exp\!\left(\mu_*+\frac{1}{2}\sigma_*^2\right).
  }
\]
Thus prediction reuses the same latent predictive mean $\mu_*$ and variance $\sigma_*^2$ derived above; only the final response map differs from the Bernoulli case.
